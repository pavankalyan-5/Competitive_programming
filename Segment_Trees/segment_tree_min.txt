#include<iostream>
#include<bits/stdc++.h>
using namespace std;

int a[100001],st[400004];
void build(int si,int ss,int se)
{
    if(ss==se)
    {
        st[si]=a[ss];
        return;
    }
    int mid=(ss+se)/2;
    build(2*si,ss,mid);
    build(2*si+1,mid+1,se);
    
    st[si]=min(st[2*si],st[2*si+1]);
}
int range(int si,int ss,int se,int qs,int qe)
{
    if(ss>qe || se<qs) return INT_MAX;
    
    if(ss>=qs && se<=qe)
    return st[si];
    
    int mid=(ss+se)/2;
    return(min(range(2*si,ss,mid,qs,qe),range(2*si+1,mid+1,se,qs,qe)));
}
int main()
{
    int n,l,r,q;
    
    cin>>n;
    
    for(int i=1;i<=n;i++) cin>>a[i];
    
    build(1,1,n);
    
    cin>>q;
    
    while(q--)
    {
        cin>>l>>r;
        cout<<range(1,1,n,l+1,r+1)<<endl;    
    }
}
